Integration Code

Under 10 lines to add GNS Auth to your website:

<!-- 1. Include the SDK -->
<script src="https://auth.globecrumbs.com/v1/gns-auth.js"></script>

<!-- 2. Add a button container -->
<div id="gns-login"></div>

<!-- 3. Initialize -->
<script>
  GNSAuth.init({
    onAuth: (res) => {
      // res.publicKey, res.signature, res.trustScore
      // Send to your server to verify
    }
  });
  GNSAuth.renderButton('#gns-login');
</script>
